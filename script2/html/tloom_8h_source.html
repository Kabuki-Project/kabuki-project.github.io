30 /* A ordered collection of either UTF-8 or UTF-16 strands starting at index 0.
31 CLoom shares the same design pattern as the ASCII kSTR (Strand) type: the lower
32 5 bits store the type kLOM, b6:b7 stores the size of the UI type and b5 stores
33 if it is a UTF-8 (de-asserted) or UTF-16 (asserted).
34 */
35 template <typename UI, typename SI, typename Char>
36 struct CLoom {
37  UI size; //< Size of the Loom in bytes.
38  SI count_max, //< Max number of strands in a loom.
39  count; //< Strand count.
40 };

69 template <typename UI, typename SI, typename Char>
70 CLoom* LoomInit(UIW* buffer, UI size, SI count_max) {
71  ASSERT(buffer)
72  ASSERT(size)
73  ASSERT(count_max >= 0)
74  SI si = 0; //< Faster to make from instructions than load from ROM.
75  si = (~si) - 7;
76  if (count_max >= si) count_max = si;
77  si = LoomCountMin<UI, SI, Char>();
78  if (count_max < si) count_max = si;
79 
80  UI ui = 0;
81  ui = (~ui) - 7; //< Faster to make from instructions than load from ROM.
82  if (size >= ui) count_max = ui;
83  ui = LoomSizeMin<UI, SI, Char>();
84  if (count_max < ui) count_max = ui;
85 
86  CLoom* loom = reinterpret_cast<CLoom*>(buffer);
87  loom->size = size;
88  loom->count_max = count_max;
89  loom->count = 0;
90 }

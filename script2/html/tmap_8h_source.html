31 /* A one-to-one map that does not use hash table.
32 
33  Maps share the same data structure as Multimaps, Dictionary(s) and Lists.
34  Maps like all Script Objects are required to are required to have a
35  stack_height, size_bytes, and header_size that is a multiple of 8 as well as
36  being .
37 
38  It is easiest to explain this data structure in terms of the Socket. Sockets
39  use a TMap to map a Slot number to a slot.
40 
41  # TMap Data Structure
42 
43  @code
44  +==========================+ -----------
45  |_____ Buffer | ^ ^
46  |_____ ... | | |
47  |_____ Data N | Data |
48  |_____ ... | | |
49  |_____ Data 0 | v |
50  |==========================| ----- |
51  |_____ count_max | ^ |
52  |_____ ... | | |
53  |_____ Sorted Mappings N | | |
54  |_____ ... | | |
55  | Sorted Mappings 1 | | |
56  |==========================| | |
57  |_____ count_max | | Size
58  |_____ ... | | |
59  |_____ Data Offset N | | |
60  |_____ ... | Header |
61  | Data Offset 1 | | |
62  |==========================| | |
63  |_____ count_max | | |
64  |_____ ... | | |
65  |_____ Type UI1 N | | |
66  |_____ ... | | |
67  | Type UI1 1 | | | ^ Up in addresses
68  |==========================| | | |
69  | TSparseArray<UI, SI> Struct | v v ^
70  +==========================+ ----------- ^ 0xN
71  @endcode
72 
73  | Map | Max Values | % Collisions (p) | Overhead |
74  |:---:|:----------:|:----------------:|:------------------------------:|
75  | 2 | 255 | 0.0001 | Ceiling (0.02*p*2^8) = 2 |
76  | 4 | 2^13 | 0.1 | Ceiling (0.04*p*2^13) = 327.68 |
77  | 8 | 2^29 | 10^-16 | Ceiling (0.04*p*2^29) = 327.68 |
78 
79  Memory Schema:
80  The memory schema is optimized for fast search and push back. When searching
81  for a key, there might or might not be a hash table.
82 
83  How to calculate size:
84  The size of any size map can be calculated as follows:
85  size = ; * (2*sizeof (SI) + sizeof (UI)) + collissionSize +
86 
87  # Cache Page Optimizations
88 
89  In order to optimize the cache pages, we need to group hot data together.
90  ChineseRoom Objs work through calling by SI, or by key by using the
91  function '\"' (i.e. "foo" is SI 44).
92 
93  # Use Case Scenario
94 
95  We are creating a plug-in API. We need to create a map in the API code, and
96  pass it over to the program. The API manages the memory for the map. This
97  map might contain several million entries, and more than 4GB of data.
98 
99  ### Why So Many TMap Types?
100 
101  We are running in RAM, and a map could contain millions of key-value pairs.
102  Adding extra bytes would added megabytes of data we don't need. Also, on
103  microcontrollers, especially 16-bit ones, will have very little RAM, so we
104  need an 16-bit object. It is easy to imagine a complex AI software using
105  more than 4GB RAM, or the need to attach a DVD ISO image as a key-value
106  pair, so we need a 64-bit map.
107 
108  # Design Strengths
109 
110  * Uses less memory.
111  * Fast push back when within buffer size.
112  * Faster inserts on small collections when within buffer size.
113  * 64-bit memory alignment ensures highest performance on Intel.
114  * Mandating zeros in buffered memory space can serve help sanitize input.
115 
116  # Design Weaknesses
117 
118  * Slow insert in large collections.
119  * Slow at growing large collections when buffer runs out.
120  * Applications with billions of items may have too much memory overhead.
121  * More complicated.
122 
123  @code

125  .
126  .
127  @endcode
128 */
129 template <typename UI = UI4, typename SI = SI4, typename I = SI2>

131  UI size; //< ASCII OBJ size.
132  SI table_size, /*< Size of the key strings in bytes.

135  Set to 0 for ASCII Map and ASCII Multimap. */
136  I count, //< Element count.
137  count_max; //< Max count of items that can fit in the socket.

159 /* Initializes a TMap from a word-aligned buffer.
160  @post Users might want to call the IsValid () function after construction
161  to verify the integrity of the object.
162  @warning The reservedNumOperands must be aligned to a 32-bit value, and it
163  will get rounded up to the next higher multiple of 4.

165 template <typename UI = UI4, typename SI = SI4, typename I = SI2>
166 TMap<UI, SI, I>* MapInit(UIW* buffer, UI size, I count_max) {
167  ASSERT(buffer)
168 
169  if (table_size >= size) return nullptr;
170  UI kSizeMin = sizeof(TMap<UI, SI, I>) +
171  count_max * (MapOverheadPerIndex<UI, SI, I>() + 2);
172  if (table_size < kSizeMin) return nullptr;
173 
174  Map<UI, SI, I>* map = reinterpret_cast<TMap*>(buffer);
175  map->size = table_size;
176  map->table_size = table_size;
177  map->size_pile = 1;
178  map->count_max = count_max;
179  map->count = 0;
180  return map;
181 }

198 /* Creates a map from dynamic memory. */
199 template <typename UI = UI4, typename SI = SI4, typename I = SI2>
200 UIW* MapNew(UI size = 0, I count_max = 0) {
201  size = AlignDown<SI8, UI>(size);
202  count_max = AlignDown<SI8, SI>(count_max);
203  UI size_min = MapSizeBoundsLower<UI, SI, I>();
204  SI count_min = MapCountBoundsLower<UI, SI, I>();
205  if (size < size_min || count_max < count_min) {
206  size = size_min;
207  count_max = size_min;
208  }
209 
210  UIW* buffer = new UIW[size >> kWordBitCount];
211  TMap<UI, SI, I>* map = reinterpret_cast<TMap<UI, SI, I>*>(buffer);
212  map->size = size;
213  map->table_size = 0;
214  map->size_pile = 0;
215  map->count = 0;
216  map->count_max = count_max;
217  return buffer;
218 }

232 /* Insets the given key-type-value tuple.
233  */
234 template <typename UI = UI4, typename SI = SI4, typename I = SI2>
235 SI MapInsert(TMap<UI, SI, I>* map, void* value, type_t type, SI index) {
236  ASSERT(map)
237  ASSERT(value)
238 
239  PRINT_LINE(id);
240 
241  SI count = map->count, count = map->stack->count, temp;
242 
243  UI table_size = map->table_size;
244 
245  if (count >= count) return ~0;
246  //< We're out of buffered indexes.
247 
248  char* types = reinterpret_cast<char*>(map) + sizeof(TMap<UI, SI, I>);
249  UI* data_offsets = reinterpret_cast<UI*>(types + count * (sizeof(UI)));
250  UI *hashes = reinterpret_cast<UI*>(types + count * (sizeof(UI) + sizeof(UI))),
251  *hash_ptr;
252  SI *indexes =
253  reinterpret_cast<SI*>(types + count * (sizeof(UI) + 2 * sizeof(SI))),
254  *unsorted_indexes = indexes + count,
255  *collission_list = unsorted_indexes + count;
256  char *keys = reinterpret_cast<char*>(map) + table_size - 1, *destination;
257 
258  // Calculate space left.
259  UI value = table_size - count * MapOverheadPerIndex<UI, SI, I>(),
260  key_length = static_cast<UI2>(strlen(id)), size_pile;
261 
262  PRINT_LINE('-');
263  PRINTF(
264  "\nAdding Key %s\n%20s: 0x%p\n%20s: %p\n%20s: 0x%p"
265  "%20s: %p\n%20s: %u\n",
266  id, "hashes", hashes, "key_offsets", key_offsets, "keys", keys, "indexes",
267  indexes, "value", value)
268 
269  UI temp_id = Hash16(id), current_mapping;
270 
271  if (key_length > value) {
272  PRINTF("\nBuffer overflow!")
273  return ~((SI)0);
274  }
275 
276  if (count == 0) {
277  map->count = 1;
278  *hashes = temp_id;
279  *key_offsets = static_cast<UI2>(key_length);
280  *indexes = ~0;
281  *unsorted_indexes = 0;
282  destination = keys - key_length;
283 
284  SlotWrite(destination, id);
285  PRINTF("\nInserted key %s at GetAddress 0x%p", id, destination)
286  MapPrint(map);
287  return 0;
288  }
289 
290  // Calculate left over buffer size by looking up last char.
291 
292  if (key_length >= value) {
293  PRINTF("\nNot enough room in buffer!")
294  return 0; //< There isn't enough room left in the buffer.
295  }
296 
297  PRINTF("\nFinding insert location...")
298 
299  SI low = 0, mid, high = count, index;
300 
301  SI* temp_ptr;
302 
303  while (low <= high) {
304  mid = (low + high) >> 1; //< Shift >> 1 to / 2
305 
306  current_mapping = hashes[mid];
307  PRINTF("\nhigh: %i mid: %i low %i hash: %x", high, mid, low,
308  current_mapping)
309 
310  if (current_mapping > temp_id) {
311  high = mid - 1;
312  } else if (current_mapping < temp_id) {
313  low = mid + 1;
314  } else {
315  PRINTF(" duplicate hash detected, ")
316 
317  // Check for other collisions.
318 
319  index = indexes[mid]; //< Index in the collision table.
320 
321  PRINTF("index:%i", (int)index)
322 
323  if (index < ~0) { //< There are other collisions.
324  PRINTF("with collisions, ")
325  // There was a collision so check the table.
326 
327  // The collisionsList is a sequence of indexes terminated
328  // by an invalid index. collissionsList[0] is
329  // an invalid index, so the collisionsList is searched from
330  // lower address up.
331  temp = indexes[mid];
332  temp_ptr = collission_list + temp;
333  index = *temp_ptr; //< Load the index in the collision table.
334  while (index < MapCountUpperBounds<SI>()) {
335  PRINTF("\ncomparing to \"%s\"", keys - key_offsets[index]);
336  if (strcmp(id, keys - key_offsets[index]) == 0) {
337  PRINTF(
338  "but table already contains key at "
339  "offset: %u.\n",
340  index);
341  return index;
342  }
343  ++temp_ptr;
344  index = *temp_ptr;
345  }
346 
347  // Its a new collision!
348  PRINTF(" new collision detected.")
349 
350  // Copy the key
351  value = key_offsets[count - 1] + key_length + 1;
352  SlotWrite(keys - value, id);
353  key_offsets[count] = value;
354 
355  // Update the collision table.
356  size_pile = map->size_pile;
357  // Shift the collisions table up one element and insert
358  // the unsorted collision index.
359  // Then move to the top of the collisions list.
360  collission_list += size_pile;
361  // and iterate down to the insert spot
362  while (collission_list > temp_ptr) {
363  *collission_list = *(collission_list - 1);
364  --collission_list;
365  }
366  *temp_ptr = count;
367 
368  map->size_pile = size_pile + 1;
369  PRINTF("\ncollision index: %u", temp)
370  // Store the collision index.
371  indexes[count] = temp; //< Store the collision index
372  map->count = count + 1;
373  hashes[count] = ~0; //< TMap the last hash to 0xFFFF
374 
375  // Move collisions pointer to the unsorted_indexes.
376  indexes += count;
377 
378  //< Add the newest char to the end.
379  indexes[count] = count;
380 
381  MapPrint(map);
382  PRINTF("\nDone inserting.")
383  return count;
384  }
385 
386  // But we still don't know if the char is a new collision.
387 
388  PRINTF("\nChecking if it's a collision... ")
389 
390  if (strcmp(id, keys - key_offsets[index]) != 0) {
391  // It's a new collision!
392  PRINTF("It's a new collision!")
393 
394  if (value < 3) {
395  PRINTF("Buffer overflow!")
396  return ~0;
397  }
398 
399  // Get offset to write the key too.
400  value = key_offsets[count - 1] + key_length + 1;
401 
402  type_t collision_index = unsorted_indexes[mid];
403  PRINTF("\n\ncollision_index: %u", collision_index)
404 
405  SlotWrite(keys - value, id);
406  PRINTF(
407  "\nInserting value: %u into index:%u "
408  "count:%u with other collision_index: %u",
409  value, index, count, collision_index)
410  key_offsets[count] = value;
411 
412  size_pile = map->size_pile;
413  indexes[mid] = static_cast<type_t>(size_pile);
414  indexes[count] = static_cast<type_t>(size_pile);
415 
416  // Insert the collision into the collision table.
417  temp_ptr = &collission_list[size_pile];
418  // Move collisions pointer to the unsorted_indexes.
419  indexes += count;
420  *temp_ptr = collision_index;
421  ++temp_ptr;
422  *temp_ptr = count;
423  ++temp_ptr;
424  *temp_ptr = ~0;
425  map->size_pile = size_pile + 3;
426  //< Added one term-type_t and two indexes.
427 
428  // Add the newest key at the end.
429  indexes[count] = count;
430 
431  // TMap the last hash to 0xFFFF
432  hashes[count] = ~0;
433 
434  map->count = count + 1;
435 
436  MapPrint(map);
437 
438  MapPrint(map);
439  PRINTF("\nDone inserting.")
440  // Then it was a collision so the table doesn't contain string.
441  return count;
442  }
443  PRINTF("\nTable already contains the key")
444  return index;
445  }
446  }
447 
448  // The id was not in the table.
449 
450  value = key_offsets[count - 1] + key_length + 1;
451  destination = keys - value;
452 
453  PRINTF(
454  "\nThe id 0x%x was not in the table so inserting %s into mid:"
455  " %i at index %u before hash 0x%x",
456  temp_id, id, mid, Diff(map, destination), hashes[mid])
457 
458  // First copy the char and set the key offset.
459  SlotWrite(destination, id);
460  key_offsets[count] = value;
461 
462  // Second move up the hashes and insert at the insertion point.
463  hash_ptr = hashes;
464  hash_ptr += count;
465  //*test = hashes;
466  PRINTF("\nl_numkeys: %u, hashes: %u hash_ptr: %u insert_ptr: %u", count,
467  Diff(map, hashes), Diff(map, hash_ptr), Diff(map, hashes + mid))
468  hashes += mid;
469  MapPrint(map);
470  while (hash_ptr > hashes) {
471  *hash_ptr = *(hash_ptr - 1);
472  --hash_ptr;
473  }
474  *hashes = temp_id;
475 
476  // Mark as not having any collisions.
477  indexes[count] = ~0;
478 
479  // Move up the sorted indexes and insert the unsorted index (which is
480  // the current count).
481  indexes += count + mid;
482  temp_ptr = indexes + count;
483 
484  while (temp_ptr > indexes) {
485  *temp_ptr = *(temp_ptr - 1);
486  --temp_ptr;
487  }
488  *temp_ptr = count;
489 
490  map->count = count + 1;
491 
492  MapPrint(map);
493  PRINTF("\nDone inserting.")
494  PRINT_LINE('-')
495 
496  return count;
497 }

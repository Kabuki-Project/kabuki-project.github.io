34 /* List of the 32 ASCII Data Types.
35  Types are organized with types that are allowed in text B-Sequences first,
36  followed by objects which get created in text using Script Operations. */
37 typedef enum AsciiTypes {
38  kNIL = 0, //< 0. kNIL/nil/void type.
39  kSI1, //< 1. 8-bit signed integer.
40  kUI1, //< 2. 8-bit unsigned integer.
41  kSI2, //< 3. 16-bit signed integer.
42  kUI2, //< 4. 16-bit unsigned integer.
43  kHLF, //< 5. 16-bit floating-point number.
44  kBOL, //< 6. 32-bit non-zero true boolean as signed integer.
45  kSI4, //< 7. 32-bit signed integer.
46  kUI4, //< 8. 32-bit unsigned integer.
47  kFLT, //< 9. 32-bit floating-point number.
48  kTMS, //< 10. 32-bit second since epoch timestamp.
49  kTSS, //< 11. 64-bit sub-second timestamp with kTMS and an kUI4 tick.
50  kTME, //< 12. 64-bit second since epoch timestamp.
51  kSI8, //< 13. 64-bit signed integer.
52  kUI8, //< 14. 64-bit unsigned integer.
53  kDBL, //< 15. 64-bit floating-point number.
54  kSIH, //< 16. 128-bit (Hexadeca-UI1) signed integer.
55  kUIH, //< 17. 128-bit (Hexadeca-UI1) unsigned integer.
56  kDEC, //< 18. 128-bit (Hexadeca-UI1) floating-point number.
57  kUIX, //< 19. 2^(6+X)-bit unsigned integer, where 0 <= X <= 7.
58  kOBJ, //< 20. N-UI1 object.
59  kADR, //< 21. UTF-8 Operand stack address.
60  kSTR, //< 22. A UTF-8 string.
61  kTKN, //< 23. A UTF-8 string token without whitespace.
62  kBSQ, //< 24. B-Sequence.
63  kLOM, //< 25. A loom of UTF-8, UTF-16, or UTF-32 strings.
64  kTBL, //< 26. A hash table.
65  kEXP, //< 27. Script^2 Expression.
66  kLST, //< 28. Set or multiset of Type-Value tuples.
67  kMAP, //< 29. One-to-one map of Integer-{Type-Value} records.
68  kBOK, //< 30. Many-to-one multimap of Key-{Type-Value} records.
69  kDIC, //< 31. One-to-one map of Key-{Type-Value} records.
70 } AsciiType;
